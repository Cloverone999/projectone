<script>
    (function(){
        // ------ DOM элементы ------
        const canvas = document.getElementById('arrayCanvas');
        const ctx = canvas.getContext('2d');
        
        // Кнопки
        const randomizeBtn = document.getElementById('randomizeBtn');
        const bubbleSortBtn = document.getElementById('bubbleSortBtn');
        const insertionSortBtn = document.getElementById('insertionSortBtn');
        const quickSortBtn = document.getElementById('quickSortBtn');
        const stopBtn = document.getElementById('stopBtn');
        const linearSearchBtn = document.getElementById('linearSearchBtn');
        const binarySearchBtn = document.getElementById('binarySearchBtn');
        const helpBtn = document.getElementById('helpBtn');
        const modal = document.getElementById('helpModal');
        const closeModalBtn = document.getElementById('closeModalBtn');
        
        // Ползунки
        const speedSlider = document.getElementById('speedSlider');
        const sizeSlider = document.getElementById('sizeSlider');
        const sizeValue = document.getElementById('sizeValue');
        const searchInput = document.getElementById('searchValue');
        const statusDiv = document.getElementById('statusMsg');
        
        // Параметры массива
        let array = [];
        let arraySize = 35;
        let isSorting = false;      // блокировка кнопок во время работы
        let stopRequested = false;   // флаг для остановки алгоритма
        let currentDelay = 40;       // задержка в мс (чем меньше, тем быстрее)
        
        // ширина столбцов и отступы
        let canvasWidth = 900;
        let canvasHeight = 400;
        let barWidth = 0;
        
        // ---------- Модальное окно (инструкция) ----------
        function openModal() {
            modal.style.display = 'flex';
        }
        function closeModal() {
            modal.style.display = 'none';
        }
        helpBtn.onclick = () => {
            openModal();
        };
        closeModalBtn.onclick = () => {
            closeModal();
        };
        // Закрытие при клике вне области контента
        modal.addEventListener('click', (e) => {
            if (e.target === modal) {
                closeModal();
            }
        });
        
        // ---------- вспомогательные функции ----------
        function updateSizeLabel() {
            sizeValue.textContent = arraySize;
        }
        
        // генерация случайного массива (значения от 5 до 95)
        function generateRandomArray() {
            if (isSorting) return;
            const newArr = [];
            for (let i = 0; i < arraySize; i++) {
                newArr.push(Math.floor(Math.random() * 85) + 10); // 10..94
            }
            array = newArr;
            drawArray(array, []);
            statusDiv.innerHTML = "🎲 Случайный массив создан";
            setTimeout(() => {
                if(!isSorting) statusDiv.innerHTML = "✅ Готов";
            }, 1200);
        }
        
        // отрисовка массива с возможностью подсветки индексов (compareIndices - массив номеров столбцов для подсветки)
        function drawArray(arr, highlightIndices = [], foundIndex = -1) {
            if (!arr.length) return;
            ctx.clearRect(0, 0, canvasWidth, canvasHeight);
            const n = arr.length;
            barWidth = (canvasWidth - 20) / n;
            const maxVal = Math.max(...arr, 100);
            
            for (let i = 0; i < n; i++) {
                let barHeight = (arr[i] / maxVal) * (canvasHeight - 60);
                if (barHeight < 4) barHeight = 4;
                const x = 10 + i * barWidth;
                const y = canvasHeight - barHeight - 20;
                
                // выбор цвета
                let color = "#3b82f6"; // базовый синий
                if (foundIndex === i) {
                    color = "#facc15"; // ярко-желтый (найденный элемент)
                } else if (highlightIndices.includes(i)) {
                    color = "#ef4444"; // красный - сравнение / обмен
                } else {
                    // градиент от бирюзового к синему
                    const intensity = 0.4 + (arr[i] / maxVal) * 0.5;
                    color = `hsl(210, 70%, ${45 + intensity * 25}%)`;
                }
                
                ctx.fillStyle = color;
                ctx.fillRect(x, y, barWidth - 1, barHeight);
                ctx.fillStyle = "#ffffffcc";
                ctx.font = "10px monospace";
                if (barWidth > 12) {
                    ctx.fillText(arr[i], x + 3, y - 3);
                }
            }
        }
        
        // задержка асинхронная
        function delay(ms) {
            return new Promise(resolve => setTimeout(resolve, ms));
        }
        
        // проверка на остановку (без дополнительной задержки, только бросок ошибки)
        async function checkStop() {
            if (stopRequested) {
                isSorting = false;
                stopRequested = false;
                statusDiv.innerHTML = "⏸️ Остановлено пользователем";
                setTimeout(() => {
                    if(!isSorting) statusDiv.innerHTML = "✅ Готов";
                }, 1500);
                throw new Error("STOPPED");
            }
            await delay(currentDelay);
        }
        
        // ---------- АЛГОРИТМЫ СОРТИРОВКИ с визуализацией ----------
        async function bubbleSortVisual() {
            let arr = [...array];
            for (let i = 0; i < arr.length; i++) {
                for (let j = 0; j < arr.length - i - 1; j++) {
                    if (stopRequested) return false;
                    drawArray(arr, [j, j+1]);
                    await delay(currentDelay);
                    if (arr[j] > arr[j+1]) {
                        [arr[j], arr[j+1]] = [arr[j+1], arr[j]];
                        drawArray(arr, [j, j+1]);
                        await delay(currentDelay);
                    }
                }
                drawArray(arr, []);
                await delay(currentDelay/2);
            }
            array = arr;
            drawArray(array, []);
            return true;
        }
        
        async function insertionSortVisual() {
            let arr = [...array];
            for (let i = 1; i < arr.length; i++) {
                let key = arr[i];
                let j = i-1;
                while (j >= 0 && arr[j] > key) {
                    if (stopRequested) return false;
                    drawArray(arr, [j, j+1]);
                    await delay(currentDelay);
                    arr[j+1] = arr[j];
                    j--;
                    drawArray(arr, [j+1]);
                    await delay(currentDelay);
                }
                arr[j+1] = key;
                drawArray(arr, [j+1]);
                await delay(currentDelay);
            }
            array = arr;
            drawArray(array, []);
            return true;
        }
        
        // быстрая сортировка (рекурсивная, но с визуализацией и остановкой)
        async function quickSortVisual(arr, left, right) {
            if (stopRequested) throw new Error("STOP");
            if (left >= right) return;
            let pivotIdx = await partition(arr, left, right);
            await quickSortVisual(arr, left, pivotIdx - 1);
            await quickSortVisual(arr, pivotIdx + 1, right);
        }
        
        async function partition(arr, left, right) {
            let pivotValue = arr[right];
            let i = left;
            for (let j = left; j < right; j++) {
                if (stopRequested) throw new Error("STOP");
                drawArray(arr, [j, right]);
                await delay(currentDelay);
                if (arr[j] <= pivotValue) {
                    [arr[i], arr[j]] = [arr[j], arr[i]];
                    drawArray(arr, [i, j]);
                    await delay(currentDelay);
                    i++;
                }
            }
            [arr[i], arr[right]] = [arr[right], arr[i]];
            drawArray(arr, [i, right]);
            await delay(currentDelay);
            return i;
        }
        
        async function runQuickSort() {
            let arr = [...array];
            try {
                await quickSortVisual(arr, 0, arr.length-1);
                if (!stopRequested) {
                    array = arr;
                    drawArray(array, []);
                }
            } catch(e) {}
            return !stopRequested;
        }
        
        // ---------- ПОИСК с анимацией ----------
        async function linearSearchVisual(target) {
            if (isSorting) return false;
            let arr = array;
            for (let i = 0; i < arr.length; i++) {
                if (stopRequested) { stopRequested=false; isSorting=false; return false; }
                drawArray(arr, [i], -1);
                await delay(currentDelay);
                if (arr[i] === target) {
                    drawArray(arr, [], i);
                    statusDiv.innerHTML = `🔎 Линейный поиск: значение ${target} НАЙДЕНО на индексе ${i}`;
                    return true;
                }
            }
            drawArray(arr, []);
            statusDiv.innerHTML = `❌ Линейный поиск: ${target} не найдено`;
            return false;
        }
        
        async function binarySearchVisual(target) {
            if (isSorting) return false;
            // для бинарного поиска массив должен быть отсортирован (сделаем копию и отсортируем без анимации, но покажем пользователю)
            let sortedArr = [...array];
            sortedArr.sort((a,b)=>a-b);
            array = sortedArr;
            drawArray(array, []);
            await delay(300);
            let left = 0, right = array.length-1;
            while (left <= right) {
                if (stopRequested) { stopRequested=false; isSorting=false; return false; }
                let mid = Math.floor((left+right)/2);
                drawArray(array, [left, right, mid], -1);
                await delay(currentDelay);
                if (array[mid] === target) {
                    drawArray(array, [], mid);
                    statusDiv.innerHTML = `🔎 Бинарный поиск: значение ${target} НАЙДЕНО на индексе ${mid}`;
                    return true;
                } else if (array[mid] < target) {
                    left = mid+1;
                } else {
                    right = mid-1;
                }
            }
            drawArray(array, []);
            statusDiv.innerHTML = `❌ Бинарный поиск: ${target} не найдено (массив отсортирован)`;
            return false;
        }
        
        // Обертка для безопасного запуска (с блокировками)
        async function startAlgorithm(algorithmFn, ...args) {
            if (isSorting) {
                statusDiv.innerHTML = "⚠️ Сначала остановите текущий алгоритм (Стоп)";
                return;
            }
            isSorting = true;
            stopRequested = false;
            statusDiv.innerHTML = "🔄 Выполнение...";
            try {
                let result;
                if (algorithmFn === linearSearchVisual || algorithmFn === binarySearchVisual) {
                    let target = parseInt(searchInput.value);
                    if (isNaN(target)) target = 42;
                    result = await algorithmFn(target);
                } else {
                    result = await algorithmFn();
                }
                if (!stopRequested && result !== false) {
                    statusDiv.innerHTML = "✅ Завершено успешно";
                } else if (!stopRequested && result === false) {
                    // ничего
                }
            } catch (err) {
                if (err.message !== "STOPPED" && err.message !== "STOP")
                    console.warn(err);
            } finally {
                isSorting = false;
                stopRequested = false;
                if(!isSorting && statusDiv.innerHTML !== "❌ ...") {
                    setTimeout(() => {
                        if(!isSorting && !statusDiv.innerHTML.includes("НАЙДЕНО") && !statusDiv.innerHTML.includes("не найдено")) 
                            statusDiv.innerHTML = "✅ Готов";
                        else if(!isSorting && (statusDiv.innerHTML.includes("НАЙДЕНО") || statusDiv.innerHTML.includes("не найдено"))) {
                            // сохраняем сообщение о поиске на некоторое время
                            setTimeout(() => {
                                if(!isSorting) statusDiv.innerHTML = "✅ Готов";
                            }, 1800);
                        } else if(!isSorting) statusDiv.innerHTML = "✅ Готов";
                    }, 1000);
                }
            }
        }
        
        // событие STOP
        function stopAll() {
            if (isSorting) {
                stopRequested = true;
                statusDiv.innerHTML = "⏸️ Остановка...";
            } else {
                statusDiv.innerHTML = "🎯 Ничего не выполняется";
                setTimeout(()=> {
                    if(!isSorting) statusDiv.innerHTML = "✅ Готов";
                },800);
            }
        }
        
        // обновление размера массива
        function resizeArray() {
            if (isSorting) {
                statusDiv.innerHTML = "Сначала остановите сортировку";
                return;
            }
            let newSize = parseInt(sizeSlider.value);
            if (isNaN(newSize)) newSize = 35;
            arraySize = newSize;
            updateSizeLabel();
            generateRandomArray();
        }
        
        function updateSpeed() {
            currentDelay = 101 - parseInt(speedSlider.value); // 1 -> 100ms задержки, 100 -> 1ms
            if (currentDelay < 1) currentDelay = 1;
        }
        
        // синхронизация размера canvas с CSS (реальные пиксели)
        function setCanvasSize() {
            const containerStyle = getComputedStyle(canvas.parentElement);
            let maxW = Math.min(1100, window.innerWidth - 60);
            canvas.width = maxW;
            canvasHeight = 400;
            canvas.height = canvasHeight;
            canvasWidth = canvas.width;
            drawArray(array, []);
        }

        randomizeBtn.onclick = () => { if(!isSorting) generateRandomArray(); else statusDiv.innerHTML="⛔ Остановите алгоритм"; };
        bubbleSortBtn.onclick = () => startAlgorithm(bubbleSortVisual);
        insertionSortBtn.onclick = () => startAlgorithm(insertionSortVisual);
        quickSortBtn.onclick = () => startAlgorithm(runQuickSort);
        stopBtn.onclick = stopAll;
        linearSearchBtn.onclick = () => startAlgorithm(linearSearchVisual);
        binarySearchBtn.onclick = () => startAlgorithm(binarySearchVisual);
        
        sizeSlider.oninput = () => {
            if(!isSorting) {
                arraySize = parseInt(sizeSlider.value);
                updateSizeLabel();
                generateRandomArray();
            } else {
                sizeSlider.value = arraySize;
            }
        };
        speedSlider.oninput = () => updateSpeed();
        window.addEventListener('resize', () => {
            setCanvasSize();
        });
        
        // инициализация
        function init() {
            arraySize = parseInt(sizeSlider.value);
            updateSizeLabel();
            updateSpeed();
            generateRandomArray();
            setCanvasSize();
        }
        
        init();
    })();
